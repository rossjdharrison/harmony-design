Code Examples

// Using Motion Variants
import { motionVariants } from './animations/motion-variants.js';

element.animate(
  motionVariants.fadeIn.keyframes,
  motionVariants.fadeIn.options
);

// Using Transition Presets
import { TRANSITIONS } from './animations/transition-presets.js';

element.style.transition = `opacity ${TRANSITIONS.NORMAL.duration}ms ${TRANSITIONS.NORMAL.easing}`;

// Respecting Reduced Motion
import { useReducedMotion } from './hooks/useReducedMotion.js';

const duration = useReducedMotion() ? 0 : 200;
element.animate([...], { duration });
