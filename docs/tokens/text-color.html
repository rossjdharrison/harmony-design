Usage Examples

CSS Custom Properties

/* Apply text colors using CSS variables */ .heading { color: var(--text-primary); } .label { color: var(--text-secondary); } .placeholder { color: var(--text-tertiary); } .disabled { color: var(--text-disabled); } .button-text { color: var(--text-inverse); }

JavaScript API

import { TEXT_COLOR_TOKENS, getTextColor } from './tokens/text-color-tokens.js'; // Get color value for current theme const primaryColor = getTextColor('text-primary', 'light'); element.style.color = primaryColor; // Apply all tokens to document import { applyTextColorTokens } from './tokens/text-color-tokens.js'; applyTextColorTokens('dark');
